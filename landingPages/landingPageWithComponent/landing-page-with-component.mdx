/// <summary>
/// Processes a credit against a credit card or token, using a predefined amount.
///   From <a href="https://help.usaepay.info/api/rest/?shell#credit-debit-open-refund">this</a> USA-ePay API link.
/// </summary>
/// <param name="_connection">USA ePay service connection instance</param>
/// <param name="amount">Amount to refund</param>
/// <param name="cardHolderName">Cardholder name</param>
/// <param name="creditCardNumber">Credit card number or token</param>
/// <param name="expirationDate">Credit card expiration date. All numbers, and needs to be formatted as MMYY.</param>
/// <param name="receiptTemplateCustomerEmail">The name of the receipt template that should be used when sending a customer receipt.</param>
/// <param name="receiptTemplateMerchantEmail">The name of the receipt template that should be used when sending a merchant receipt.</param>
/// <param name="software">Software name and version (useful for troubleshooting)</param>
public static async Task<ProviderResponse<TransactionsResponse>> CreditCardCredit(IUSAePayConnection _connection,
    decimal amount, string cardHolderName, string creditCardNumber, string expirationDate, string receiptTemplateCustomerEmail,
    string receiptTemplateMerchantEmail, string software)
{
    var transactionsBody = new TransactionsRequest
    {
        Amount = amount,
        Command = "cc:credit",
        CreditCard = new TransactionsCreditCard
        {
            CardHolder = cardHolderName,
            Number = creditCardNumber,
            Expiration = expirationDate,
        },
        ReceiptTemplateCustomerEmail = receiptTemplateCustomerEmail,
        ReceiptTemplateMerchantEmail = receiptTemplateMerchantEmail,
        Software = software
    };
    var resp = await _connection.PostTransactions(transactionsBody);
    return resp;
}
